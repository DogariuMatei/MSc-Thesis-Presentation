\begin{frame}{Self-replicating infrastructure}
  Inspired by von Neumann's\cite{vonneumann} theory of cellular automata, we set out to build the first self-replicating system that runs on \textbf{real} commercial hardware.
  \begin{itemize}
    \item \textbf{Self-replication}: the ability of an entity to autonomously reproduce. 
    \item \textbf{In our case}: spawn identical clones that use the same software and run on bought hardware. Self-scaling network.
  \end{itemize}
  \medskip
  \medskip
  \medskip

\end{frame}

{
\begin{frame}{Use case}
    \medskip
    CDN's are operated by humans who serve the entity, rather than the content owners.
    \begin{itemize}
        \item Traditional CDN's concentrate operational control and content availability.
        \item We decentralize content, relinquish control, and replace humans with protocol.
    \end{itemize}
    To put it simply, we created an \textbf{alternative to YouTube's services}, by changing the delivery mechanism for Creative Commons content.
    \medskip
    \medskip
    \medskip
    \begin{block}{From theory to real-world service}
        Self-replication is just the technical underpinning of the overarching goal: liberating mainstream content delivery networks.
    \end{block}
    
\end{frame}
}


{
\begin{frame}{Just a P2P Network?}
    \medskip
    \textbf{Yes}, but it is complicated. We propose a network of \textbf{self-replicating seedbox servers}.
    \begin{itemize}
        \item Peer-to-peer networks distribute content
      across nodes, but the hardware is still managed by humans.
        \item We want to replace human oversight and control with \textbf{autonomous software agents}.
    \end{itemize}

    \medskip
    \medskip
    \medskip
    \begin{alertblock}{The open research gap}
    Can the \textbf{infrastructure layer itself} be operator-independent, with no human controlling it after genesis?
    \end{alertblock}
    
\end{frame}
}


{
  \setbeamerfont{itemize/enumerate body}{size=\small}
  \begin{frame}{Self-replicating decentralized CDN infrastructure}
    We propose a network of \textbf{self-replicating seedbox servers} that: 
    \medskip
    \begin{itemize}
        \item Run on commercial VPS instances.
        \item Rent the hardware themselves using Bitcoin.
        \item Seed Creative Commons YouTube content over BitTorrent.
        \item Can spawn children autonomously.
        \item Only human action is bootstrapping the genesis node.
    \end{itemize}

    \medskip
    \textcolor{tud Orange}{\textbf{The research gap is filled --- but can such a
    fleet actually \emph{survive} and \emph{grow} on its own?}}
  \end{frame}
}

  \begin{frame}{Research question}
    \begin{columns}[T]
      \begin{textcolumn}
        \color{white}\large
        Can a fleet of autonomous software agents, controlled only by inherited economic parameters, \textbf{survive} and \textbf{grow} without further human
        intervention?
      \end{textcolumn}
      \begin{textcolumn}
        \textbf{The main hypothesis}: network

        \medskip

\begin{frame}{Research Novelty}
    \small
    We propose three contributions:
      \begin{enumerate}
        \item \textbf{Engineering} --- first self-replicating economic agent that buys its own infrastructure on an open market.
        \item \textbf{Testing Method} --- unmodified production code run against a replica Bitcoin network and a mock VPS marketplace.
        \item \textbf{Empirical} --- fleet dynamics and one heritable trait under
          selection.
      \end{enumerate}

      \textcolor{tud Orange}{\textbf{Research Novelty:}} first self-replicating system on \textbf{real commercial infrastructure paid for with real money}.
\end{frame}
