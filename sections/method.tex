  \renewcommand{\sectionsubtitle}{Testing the node code offline and without cost}
  \section{Methodology}
}


{
\setbeamerfont{itemize/enumerate body}{size=\small} % template pins lists to \normalsize
\begin{frame}{Faithful exercise of production code}
  \small
  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.44\textwidth}
      \begin{block}{Unchanged from live deployment}
        Node code is executed \textbf{identically} to a real deployment. Only changes relate to the compressed-time intervals: 1 sim-day \(\approx\) \qty{22}{s}.
      \end{block}
    \end{column}
    \begin{column}{0.52\textwidth}
    \smallskip
    \smallskip
      \textbf{Swapped the network layer only:}
      \begin{itemize}
        \item Bitcoin \(\to\) private \textbf{regtest}, node reuses same paths.
        \item SporeStack API \(\to\) host-side \textbf{mock}, identical endpoints.
          Invoices paid in real regtest BTC.
        \item Each ``server'' \(=\) a real \textbf{Alpine LXC container}. Parent deploys over the production SSH pipeline.
        \item Peer discovery via the same IPv8 tracker. An IPv8 bootstrap server keeps the fleet offline.
      \end{itemize}
    \end{column}
  \end{columns}
  \smallskip
  \smallskip
  \smallskip
  \smallskip
  \textcolor{tud Orange}{\textbf{What runs locally is the code that would run on a live deployment.}}
\end{frame}
}


{
\setbeamerfont{itemize/enumerate body}{size=\small} % template pins lists to \normalsize
\begin{frame}{Two experiments}
  \small
  \begin{columns}[T, onlytextwidth]
    \begin{column}{0.48\textwidth}
    \smallskip
      \textbf{No-income run} --- closed system
      \begin{itemize}
        \item Genesis seeded with  \textbf{only €10,000}.
        \item Every node spends only its birth runway \(+\) inheritance.
        \item A \textcolor{tud Orange}{falsifiable closed-system prediction}: the
          fleet cannot sustain itself indefinitely.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
    \smallskip
      \textbf{Income run} --- open system
      \begin{itemize}
        \item Mock income pays the fleet
        \item Per-node income \(\approx\)
          invariant (scales with node count).
        \item Uneven daily shares.
        \item \textbf{Node cap}: income suspended at 60 nodes, resumed at 40
          \(\to\) alternating growth/drought.
      \end{itemize}
    \end{column}
  \end{columns}
 \smallskip
 \smallskip
 \smallskip
  \vspace{-1ex}
  \begin{alertblock}{One caveat}
    Under load, the regtest drops \(\sim\)half of all broadcasts, so income
    is set at \textbf{\(\sim\)10\(\times\) per-node rent}. \textbf{The income run tests fleet mechanics under favorable
    conditions, not its economic viability.}
  \end{alertblock}
\end{frame}
}
